rule ::= symbol '::=' expression

symbol ::= [a-zA-Z_] ([a-zA-Z0-9_])*

expression ::= term ('|' term)*

term ::= '('? cterm '-' cterm ')'? quantifier?

cterm ::= (hexChar | charSet | literal)+

hexchar ::= '#x' [0-9a-f]+

charset ::= '[' '^'? (charrange | char)+ ']'

charrange ::= hexchar '-' heaxhar | char '-' char

char ::= [^-]]

literal ::=

quantifier ::= '?' | '+' | '*'
